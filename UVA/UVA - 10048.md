#include <iostream>
#include <bits/stdc++.h>
#define maxn 101
#define INF 0x3f3f3f3f
using namespace std;
struct Edge
{
    int u,v,w;
    Edge(int i,int j,int k):u(i),v(j),w(k){}
    Edge(){}
};
int n,m,q;
int d[maxn][maxn];//ç‚¹
vector<Edge>edges;

void init()
{
    edges.clear();
    for(int i = 1;i<=n;i++)
    {
        for(int j = 1;j<=n;j++)
            d[i][j] = INT_MAX;
        d[i][i] = 0;
    }
}

void floyd()
{
    for(int k = 1;k<=n;k++)
    {
        for(int i = 1;i<=n;i++)
        {
            for(int j = 1;j<=n;j++)
            {
                if( d[i][k]<d[i][j]&&d[k][j]<d[i][j] )
                {
                    d[i][j] = max(d[i][k],d[k][j]);
                }
            }
        }
    }
}


int main() {
    int u,v,w;
    int kase = 0;
    while (cin>>n>>m>>q&&n&&m&&q)
    {
        init();
        for(int i = 0;i<m;i++)
        {
            cin>>u>>v>>w;
            d[u][v] = w;
            d[v][u] = w;
        }
        floyd();
        if(kase++) cout<<endl;
        printf("Case #%d\n",kase);
        for(int i = 0;i<q;i++)
        {
            cin>>u>>v;
            if(d[u][v]==INT_MAX)
                cout<<"no path"<<endl;
            else
                cout<<d[u][v]<<endl;
        }
    }
    return 0;
}
